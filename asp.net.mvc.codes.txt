// now we add delete js function 

function onDelete(url){
	swal({
		title : "آیا مطمئن هستید؟",
		text  : "این عملیات تغییر ناپذیر است",
		icon  : "warning",
		buttons: true , 
		dangerMode : true
	}).then((willDelete) => {
		if(willDelete){
			$.ajax({
				type : "Delete",
				url  : url ,
				success : function(data){
					if(data.success){
						toastr.success(data.message);
						datatable.ajax.reload();
					}else{
						toastr.error(data.message);
					}			
				}
			});
		}
	});
}
